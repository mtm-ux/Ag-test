    <script>
      const Antigravity = {
        toggle: function(url, containerId, themeClass) {
          const container = document.getElementById(containerId);
          const viewport = container.querySelector('iframe');
          
          // Wenn das gleiche Spiel nochmals geklickt wird -> Schließen
          if (viewport.getAttribute('data-current-url') === url && container.classList.contains('active')) {
            container.classList.remove('active');
            setTimeout(() => { viewport.src = "about:blank"; viewport.setAttribute('data-current-url', ''); }, 600);
          } else {
            // Neues Spiel laden
            viewport.src = url;
            viewport.setAttribute('data-current-url', url);
            viewport.className = `app-viewport ${themeClass}`;
            container.classList.add('active');
            // Sanftes Scrollen zum Spiel
            setTimeout(() => {
              container.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }, 300);
          }
        }
      };
    </script>
